**3. Add a helper function to format audit log descriptions**

Define inside the component, before the return:
```typescript
function formatAuditAction(log: AuditLog): string {
  const meta = log.metadata || {}
  const actor = (meta.actor_name as string) || (meta.email as string) || 'Admin'
  const target = (meta.target_name as string) || (meta.email as string) || log.target_id || 'Unknown'
  const role = (meta.new_role as string) || (meta.role as string) || ''

  switch (log.action) {
    case 'user.invite':
      return t('audit_log.action_invite', { actor, target: (meta.email as string) || target, role })
    case 'user.role_change':
      return t('audit_log.action_role_change', { actor, target, role: (meta.new_role as string) || role })
    case 'user.deactivate':
      return t('audit_log.action_deactivate', { actor, target })
    case 'user.reactivate':
      return t('audit_log.action_reactivate', { actor, target })
    default:
      return `${actor}: ${log.action}`
  }
}
```

**4. Add the action pill color helper**

Define inside the component:
```typescript
function auditActionPill(action: string): string {
  switch (action) {
    case 'user.invite':
      return 'bg-brand-accent/15 text-brand-accent'
    case 'user.role_change':
      return 'bg-brand-orange/15 text-brand-orange'
    case 'user.deactivate':
      return 'bg-brand-red/15 text-brand-red'
    case 'user.reactivate':
      return 'bg-brand-green/15 text-brand-green'
    default:
      return 'bg-gray-100 text-gray-600'
  }
}

function auditActionLabel(action: string): string {
  switch (action) {
    case 'user.invite': return 'Invite'
    case 'user.role_change': return 'Role Change'
    case 'user.deactivate': return 'Deactivate'
    case 'user.reactivate': return 'Reactivate'
    default: return action
  }
}
```
