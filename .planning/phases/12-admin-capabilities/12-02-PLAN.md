The payments data is already fetched (`allPayments`). Compute MRR from ALL payments (not just the 50 already limited), so add a SEPARATE query for MRR that fetches last 12 months of paid payments:

```typescript
// MRR data — last 12 months of paid payments (per D-19)
const now = new Date();
const twelveMonthsAgo = new Date(now.getFullYear(), now.getMonth() - 11, 1);
const { data: paidPayments } = await supabase
  .from("payments")
  .select("amount, paid_at, created_at")
  .eq("status", "paid")
  .gte("created_at", twelveMonthsAgo.toISOString())
  .order("created_at", { ascending: true });

// Group by month
const mrrByMonth: Record<string, number> = {};
for (const p of paidPayments || []) {
  const dateStr = p.paid_at || p.created_at;
  const key = dateStr.slice(0, 7); // "YYYY-MM"
  mrrByMonth[key] = (mrrByMonth[key] || 0) + (p.amount || 0);
}

// Build 12-month array with gaps filled as 0
const months: { month: string; mrr: number }[] = [];
for (let i = 11; i >= 0; i--) {
  const d = new Date(now.getFullYear(), now.getMonth() - i, 1);
  const key = `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, "0")}`;
  const label = d.toLocaleDateString("en-US", { month: "short", year: "2-digit" });
  months.push({ month: label, mrr: mrrByMonth[key] || 0 });
}
setMrrData(months);
```
